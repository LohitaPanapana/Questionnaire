public class Largest{ 
public static int thirdLargest(int[] a, int total){ 
int temp; 
for (int i = 0; i < total; i++) 
{ 
for (int j = i + 1; j < total; j++) 
{ 
if (a[i] > a[j]) 
{ 
temp = a[i]; 
a[i] = a[j]; 
a[j] = temp; 
} 
} 
} 
return a[total-3]; 
} 
public static void main(String args[]){ 
int a[]={1,2,4,8,9,2}; 
System.out.println("Third Largest: "+thirdLargest(a,6));

}
}
